115 void get_animations(STRING *file, STRING *character, ANIMATION *anim) {
116  var filehandle_n = file_open_read(file);
117 
118  // Print error on screen if the character that has to be loaded does not exist in the file
119  if(file_find(filehandle_n, character) < 0) { // get character
120  STRING *errorStr = str_create("");
121  str_cpy(errorStr, "the NPC the you are looking for (");
122  str_cat(errorStr, character);
123  str_cat(errorStr, ") does not exist in the file (");
124  str_cat(errorStr, file);
125  str_cat(errorStr, ").");
126 
127  printf(_chr(errorStr), NULL);
128  }
129 
130  file_find(filehandle_n, "animations: "); // get animation handle
131 
132  STRING* animStr = str_create("");
133  file_str_read(filehandle_n, animStr); // animation should be divided with spaces... (ex: idle run walk jump swim)
134 
135  file_close(filehandle_n);
136 
137  var animStrLength; // stores the next animation name
138  var curAnimStrLength;
139 
140  STRING* currentAnim = str_create("");
141  STRING* findAnim = str_create("");
142  STRING* getAnim = str_create("");
143 
144  var valueLength;
145  var lengthClipValue;
146  var animValueArray[3];
147  var i;
148 
149  STRING* valueStr = str_create("");
150 
151  LINKED_ANIM *previous = init_linkedAnimation("empty", 0, 0, 0);
152  LINKED_ANIM *current;
153 
154  while(1) {
155  // find new animations string
156  animStrLength = str_stri(animStr, " "); // check for the next space seperating the animations
157  if(animStrLength == 0) { break; } // if no more animations are found, break this loop.
158 
159  // process animation into new game animation
160  str_cpy(currentAnim, animStr); // store animations string
161  str_clip(animStr, animStrLength); // remove animation from animations string
162  curAnimStrLength = str_len(animStr);
163  str_trunc(currentAnim, curAnimStrLength+1); // remove all but first animation name (and the extra space)
164 
165  filehandle_n = file_open_read(file);
166 
167  // add new struct for linked animation list here
168  str_cpy(findAnim, currentAnim);
169  str_cat(findAnim, " animation: "); // make animation name
170 
171  file_find(filehandle_n, character); // get character
172  file_find(filehandle_n, findAnim); // find animation
173 
174  file_str_read(filehandle_n, getAnim); // get parameters
175 
176  file_close(filehandle_n);
177 
178  for(i=0; i<3; i++){
179  // get the animation handle
180  switch(i) {
181  case 0:
182  valueLength = str_stri(getAnim, "blend=");
183  lengthClipValue = str_len("blend");
184 
185  break;
186 
187  case 1:
188  valueLength = str_stri(getAnim, "animspeed=");
189  lengthClipValue = str_len("animspeed");
190  break;
191 
192  case 2:
193  valueLength = str_stri(getAnim, "blendspeed=");
194  lengthClipValue = str_len("blendspeed");
195  break;
196  }
197 
198  str_clip(getAnim, valueLength+lengthClipValue);
199 
200  // get value now
201  valueLength = str_stri(getAnim, " ");
202  str_cpy(valueStr, getAnim); // store animations string
203 
204  valueLength = str_len(valueStr);
205  str_trunc(valueStr, valueLength+1);
206 
207  animValueArray[i] = str_to_num(valueStr);
208  }
209 
210  current = init_linkedAnimation(currentAnim, animValueArray[0], animValueArray[1], animValueArray[2]); // get first animation
211  if(!str_cmpi(previous->ANIM_animName, "empty")) { previous->ANIM_next = current; } else { anim->ANIM_first = current; } // add first linked animation into the player
212  previous = current; // link current to previous to inherit for the next animation set.
213  }
214 
215  sys_free(previous);
216  sys_free(current);
217 }
